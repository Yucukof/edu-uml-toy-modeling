\newglossaryentry{Use}{name=Use,description={Action d'utiliser un objet consommable de type \textit{Food} ou \textit{Drink} présent l'inventaire. L'objet consommé effectue l'action de son type, puis décompté.\newline\textit{synonymes} : utiliser, activer, employer}}
\newglossaryentry{Useleft}{name=Use Left,description={Un indicateur du nombre d'utilisation possibles restantes pour un objet donné dans l'inventaire. Une fois que ce nombre atteint zéro, l'objet ne peut plus être utilisé.\newline\textit{synonymes} : point d'utilisation}}
\newglossaryentry{Variable}{name=Variable,description={Une entité nommée, associée à un type de données et à laquelle on va pouvoir affecter une valeur}}
\newglossaryentry{VariableGlobal}{name=Variable: Global,description={Se dit d'une variable déclarée par une stratégie et qui peut être utilisée par un module déclaré sous cette même stratégie.\newline\textit{synonymes} : variable globale}}
\newglossaryentry{VariableGlobal}{name=Variable: Local,description={Se dit d'une variable déclarée par unun module et étant accessible uniquement à ce module.\newline\textit{synonymes} : variable locale}}
\newglossaryentry{Wall}{name=Wall,description={Un objet indestructible, servant notamment à démarquer la frontière du plateau.\newline\textit{synonymes} : mur}}
\newglossaryentry{Water}{name=Water,description={Un obstacle uniquement franchissable si le franchisseur possède le masque de plongée.\newline\textit{synonymes} : (Zone d')eau, bassin}}
\newglossaryentry{Wear}{name=Wear,description={Action d'utiliser un objet de type \textit{Buff} présent dans l'inventaire.\newline Déclenche le décompte d'utilisation. Une fois le décompte achevé, l'objet est décompté de l'inventaire et le personnage retiré.\newline\textit{synonymes} : porter, enfiler, activer}}
\newglossaryentry{Zombie}{name=Zombie,description={Un personnage non-jouable, présent sur le plateau de jeu et cherchant à attaquer et tuer les joueurs indistinctement. Il ne possède ni inventaire ni points de défense.\newline\textit{synonymes} : zombie, mob, ennemi, monstre}}
